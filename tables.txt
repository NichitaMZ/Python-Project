CREATE TABLE CUSTOMER (
cust_id varchar(10) PRIMARY KEY NOT NULL,
cust_name varchar(20) NOT NULL,
cust_surname varchar(20) NOT NULL,
cust_locality varchar(25), (not null)
cust_street varchar(20),
cust_house_no varchar(3),
cust_postcode varchar(8),
cust_mob varchar(12),
cust_email varchar(35) 
);



CREATE TABLE PRODUCT (
prod_id varchar(10) PRIMARY KEY NOT NULL,
supplier_id varchar(10),
FOREIGN KEY (supplier_id) references suppliers(supplier_id),
prod_name varchar(25),
prod_colour varchar(15),
prod_size varchar(3),
prod_qty int,
prod_sprice double
);



CREATE TABLE sales(
sale_id int,
sale_date Date,
cust_id varchar(10),
FOREIGN KEY (cust_id) references customer(cust_id),
prod_id varchar(10),
FOREIGN KEY (prod_id) references product(prod_id),
sale_qty int,
CONSTRAINT chk_sale
PRIMARY KEY(sale_id,prod_id)
);
 




CREATE TABLE SUPPLIERS (
supplier_id(10) PRIMARY KEY NOT NULL,
supplier_name(30),
supplier_country(15),
supplier_email(30),
supplier_contact_no int
);
